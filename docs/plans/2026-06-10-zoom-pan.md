> **Prerequisite:** Plan 1 (extraction) is merged — these changes land in the `aeye` repo, against `gallery.go` / `gallery_render.go` / `gallery_cache.go`.

**Goal:** Add zoom (`z`/`Z`/`0`) and panning (arrow keys) to the carousel preview on kitty-graphics terminals.

**Architecture:** kitty renders the preview by transmitting an image scaled to a fixed cell box (Unicode-placeholder protocol). **Zoom = crop the source image in Go to a sub-rectangle and let the same cell box magnify it.** The currently-selected image is decoded once into memory; each zoom/pan keystroke crops + downscales from that in-memory image to a fixed scratch PNG and re-transmits only the preview (not the filmstrip). Zoom/pan state survives the 1.5s auto-refresh tick; it resets only when the selection changes. chafa fallback ignores zoom this phase.

**Tech Stack:** Go (`golang.org/x/image/draw` for crop+scale, already a dependency), kitty graphics protocol.

package main

import "image"

// clampF clamps a float64 to [lo, hi].
func clampF(v, lo, hi float64) float64 {
	if v < lo {
		return lo
	}
	if v > hi {
		return hi
	}
	return v
}

// cropRect returns the source-pixel rectangle to display at the given zoom and
// pan. zoom >= 1 (1 = the whole image). panX/panY are the crop *center* as a
// fraction (0..1) of the source; the origin is clamped so the crop stays within
// the source bounds. The returned rect is in source coordinates with origin
// (0,0) — callers offset by the image's Bounds().Min before sampling.
func cropRect(srcW, srcH int, zoom, panX, panY float64) image.Rectangle {
	if zoom < 1 {
		zoom = 1
	}
	cw := int(float64(srcW) / zoom)
	ch := int(float64(srcH) / zoom)
	if cw < 1 {
		cw = 1
	}
	if ch < 1 {
		ch = 1
	}
	cx := clamp(int(panX*float64(srcW))-cw/2, 0, srcW-cw)
	cy := clamp(int(panY*float64(srcH))-ch/2, 0, srcH-ch)
	return image.Rect(cx, cy, cx+cw, cy+ch)
}
```

(`clamp(int,int,int)` already exists in `gallery.go`.)

const zoomMax = 8.0

func (m *galleryModel) resetZoom() {
	m.zoom, m.panX, m.panY = 1, 0.5, 0.5
}

// zoomBy multiplies the zoom by factor, clamped to [1, zoomMax]. Reaching 1
// recenters the pan (a fit image has no meaningful pan).
func (m *galleryModel) zoomBy(factor float64) {
	m.zoom = clampF(m.zoom*factor, 1, zoomMax)
	if m.zoom == 1 {
		m.panX, m.panY = 0.5, 0.5
	}
}

// panBy shifts the crop center. The step is divided by zoom so it feels like a
// constant on-screen distance regardless of magnification. cropRect clamps the
// actual origin, so panX/panY are loosely clamped to [0,1] here.
func (m *galleryModel) panBy(dx, dy float64) {
	m.panX = clampF(m.panX+dx/m.zoom, 0, 1)
	m.panY = clampF(m.panY+dy/m.zoom, 0, 1)
}
```

- [ ] **Step 1: Implement `ensureDecoded`** in `gallery_zoom.go`

```go
import (
	"image"
	"os"
)

// ensureDecoded decodes the currently-selected image into m.curImg, but only
// when the selected path changed since the last decode. A changed selection
// resets zoom/pan to fit; an unchanged selection (e.g. an auto-refresh tick that
// appended a different image) preserves the current zoom/pan and the decode.
func (m *galleryModel) ensureDecoded() {
	if len(m.images) == 0 {
		m.curImg, m.curImgPath = nil, ""
		return
	}
	p := m.images[m.cursor].Path
	if p == m.curImgPath && m.curImg != nil {
		return
	}
	m.curImgPath = p
	m.resetZoom()
	f, err := os.Open(p)
	if err != nil {
		m.curImg = nil
		return
	}
	defer f.Close()
	img, _, err := image.Decode(f)
	if err != nil {
		m.curImg = nil
		return
	}
	m.curImg = img
}
```

(The `image/jpeg`, `image/gif`, `golang.org/x/image/webp` decoders are already registered via blank imports in `gallery_cache.go`, so `image.Decode` handles all manifest formats.)

```go
import (
	"os"
	"path/filepath"

	"golang.org/x/image/draw"
)

var zoomScratch = filepath.Join(os.TempDir(), "aeye-zoom.png")

// renderZoom crops m.curImg per (zoom, pan), downscales the crop to the
// cols×rows cell box, writes it to a fixed scratch PNG, and returns that path.
// Returns the raw selected path when nothing is decoded or zoom <= 1 (so the
// unzoomed path is byte-for-byte the pre-zoom behavior).
func (m *galleryModel) renderZoom(cols, rows int) string {
	raw := m.images[m.cursor].Path
	if m.curImg == nil || m.zoom <= 1 {
		return raw
	}
	b := m.curImg.Bounds()
	r := cropRect(b.Dx(), b.Dy(), m.zoom, m.panX, m.panY).Add(b.Min)
	tw, th := cols*cellPxW, rows*cellPxH
	scale := min(float64(tw)/float64(r.Dx()), float64(th)/float64(r.Dy()))
	if scale > 1 {
		scale = 1 // never upscale past source resolution
	}
	dst := image.NewRGBA(image.Rect(0, 0, int(float64(r.Dx())*scale), int(float64(r.Dy())*scale)))
	draw.CatmullRom.Scale(dst, dst.Bounds(), m.curImg, r, draw.Src, nil)
	return writePNG(zoomScratch, dst, raw)
}
```

(`cellPxW`/`cellPxH` and `writePNG` already exist in `gallery_cache.go`; `min` is a Go 1.21+ builtin.)

import "fmt"

// deletePreview clears just the preview image (id=previewID) + its placements,
// leaving the filmstrip untouched — so zoom/pan keystrokes don't re-transmit the
// whole grid (which would flicker).
func deletePreview() string {
	return tmuxPassthrough(fmt.Sprintf("\x1b_Ga=d,d=i,i=%d,q=2\x1b\\", previewID))
}

// transmitPreviewOnly re-renders + re-transmits only the preview image at the
// current zoom/pan. The placeholder cells in View already reference previewID.
func (m *galleryModel) transmitPreviewOnly() {
	if m.backend != backendKitty || m.tty == nil || len(m.images) == 0 {
		return
	}
	fmt.Fprint(m.tty, deletePreview())
	fmt.Fprint(m.tty, transmitVirtual(previewID, m.renderZoom(m.l.previewW, m.l.previewH), m.l.previewW, m.l.previewH))
}
